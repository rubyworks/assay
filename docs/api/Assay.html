def self.failure_classes_by_operator
  @_failure_classes_by_operator ||= (
    c = {}
    ObjectSpace.each_object(Class) do |fc|
      next unless fc < Assay::Assertion
      if fc.respond_to?(:assertion_operator)
        c[fc.assertion_operator.to_sym] = fc
      end
    end
    c
  )
end
